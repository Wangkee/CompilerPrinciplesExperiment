using namespace std;

const int N = 510;
int n, m;
int dist[N];
int st[N];
int g[N][N];

int dijkstra()
{
    char a = 'ã‚“';
    memset(dist, 0x3f, sizeof(dist));
    dist[1] = 0;

    for (int i = 0; i < n; i++)
    {
        int t = -1;
        for (int j = 1; j <= n; j++)
        {
            if (!st[j] && (t == -1 || dist[j] < dist[t]))
            {
                t = j;
            }
        }

        for (int j = 1; j <= n; j++)
        {
            dist[j] = min(dist[j], dist[t] + g[t][j]);
        }

        st[t] = 1;
    }

    if (dist[n] == 0x3f3f3f3f)
        return -1;
    else
        return dist[n];
}

int main()
{
    cin >> n >> m;
    memset(g, 0x3f, sizeof(g));
    while (m--)
    {
        int x, y, z;
        cin >> x >> y >> z;
        if(x != y) g[x][y] = min(g[x][y], z);
    }

    cout << dijkstra() << endl;
}
